# 1. Go to project folder (only if needed)
cd /path/to/your/project

# 2. Make sure you're on the right branch
git status
git switch main     # or: git checkout main

# 3. Get latest changes from GitHub
git pull origin main

# 4. See changed files
git status

# 5. Stage all changes
git add .
# (or stage specific files)
# git add src/controllers/sign_in.ts

# 6. Commit with a message
git commit -m "Add signIn controller with bcrypt.compare"
 
# 7. Push your commit(s) to GitHub
git push origin main

summary of daily work flow is:
git pull origin main   # always first
git status(optional but good)
git add .(or git add specific_file)
git commit -m "Your message"
git push origin main
